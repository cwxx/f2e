## 问题

```js
function* gen(a) {
  let length = a.length;
  for (let i = 0; i < length; i++) {
    let item = a[i];
    if (typeof item !== 'number') {
      yield* gen(item);
    } else {
      yield item;
    }
  }
}

function main() {
  const tasks = gen([[[4], 5, 6], [1], 0]);
  let result;
  while (result = tasks.next(), result.done !== true) {
    console.log(result.value)
  }
}

main();
```
